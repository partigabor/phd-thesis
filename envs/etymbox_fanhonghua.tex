\begin{etymology}\label{ety:fanhonghua}
\textbf{Mandarin Chinese} \traditionalchinesefont{番紅花} \textit{fānhónghuā} `saffron' [foreign-red-flower], reached China from way of Kashmir and Tibet, hence the synonyms \traditionalchinesefont{藏紅花} \textit{zànghónghuā} [Tibetan-red-flower], \traditionalchinesefont{西紅花} \textit{xīhónghuā} [western-red-flower]\footnote{\textcite{kleeman_oxford_2010}}
\end{etymology}
