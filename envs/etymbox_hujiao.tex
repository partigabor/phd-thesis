\begin{etymology}\label{ety:hujiao}
\textbf{Mandarin Chinese} \traditionalchinesefont{胡椒} \textit{hú​jiāo} `black pepper' [barbarian-pepper], from \traditionalchinesefont{胡} \textit{hú​} `Western barbarians, steppe nomads' + \traditionalchinesefont{椒} \textit{jiāo} `pepper, spice' (\textit{jiāo} was the prototype spice in China, originally referring to the local ``Sichuan pepper'' which is now called \traditionalchinesefont{花椒} \textit{huājiāo} [flower-pepper]), [Northern and Southern] 420-445\footnote{\textcite{schuessler_abc_2007}}
\end{etymology}
